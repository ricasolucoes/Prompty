    ```ts
    import { createClient } from '@/lib/supabase/server'

    export async function getProfileByUsername(username: string) {
      const supabase = await createClient()
      const { data, error } = await supabase
        .from('profiles')
        .select('id, username, display_name, avatar_url, bio, points, level, created_at')
        .eq('username', username)
        .maybeSingle()
      if (error) return null
      return data
    }

    export async function getPublishedPromptysByAuthor(authorId: string) {
      const supabase = await createClient()
      const { data, error } = await supabase
        .from('promptys')
        .select('id, title, description, cover_image_url, status, created_at')
        .eq('author_id', authorId)
        .eq('status', 'published')
        .order('created_at', { ascending: false })
      if (error) return []
      return data ?? []
    }

    export async function getOwnPromptys() {
      const supabase = await createClient()
      const { data: { user } } = await supabase.auth.getUser()
      if (!user) return []
      const { data, error } = await supabase
        .from('promptys')
        .select('id, title, status, created_at, updated_at')
        .eq('author_id', user.id)
        .order('updated_at', { ascending: false })
      if (error) return []
      return data ?? []
    }
    ```

    Create `src/lib/storage/avatar.ts` (used from client component — uses browser supabase client):
    ```ts
    import { createClient } from '@/lib/supabase/client'

    /**
     * Uploads a webp avatar to prompty-images/{userId}/avatar.webp.
     * Caller MUST pass a compressed File (use compressAvatar first).
     * Path-scoped RLS: (storage.foldername(name))[1] === auth.uid()::text
     */
    export async function uploadAvatar(file: File, userId: string): Promise<string> {
      if (file.type !== 'image/webp') throw new Error('Avatar must be webp')
      if (file.size > 200 * 1024) throw new Error('Avatar must be <=200KB')
      const supabase = createClient()
      const path = `${userId}/avatar.webp`
      const { error } = await supabase.storage
        .from('prompty-images')
        .upload(path, file, { upsert: true, contentType: 'image/webp', cacheControl: '3600' })
      if (error) throw error
      const { data } = supabase.storage.from('prompty-images').getPublicUrl(path)
      // Cache-bust so the browser fetches the new image after upload
      return `${data.publicUrl}?v=${Date.now()}`
    }
    ```

    ```tsx
    'use client'
    import { useState } from 'react'
    import Image from 'next/image'
    import { toast } from 'sonner'
    import { compressAvatar } from '@/lib/images/compress'
    import { uploadAvatar } from '@/lib/storage/avatar'

    export function AvatarUpload({ userId, initialUrl, onChange }: {
      userId: string
      initialUrl: string | null
      onChange: (url: string) => void
    }) {
      const [url, setUrl] = useState<string | null>(initialUrl)
      const [busy, setBusy] = useState(false)

      async function handleFile(e: React.ChangeEvent<HTMLInputElement>) {
        const file = e.target.files?.[0]
        if (!file) return
        if (file.size > 2 * 1024 * 1024) { toast.error('Imagem maior que 2MB'); return }
        setBusy(true)
        try {
          const compressed = await compressAvatar(file)
          const publicUrl = await uploadAvatar(compressed, userId)
          setUrl(publicUrl)
          onChange(publicUrl)
          toast.success('Avatar atualizado')
        } catch (err: any) {
          toast.error(err.message || 'Falha ao enviar avatar')
        } finally {
          setBusy(false)
        }
      }

      return (
        <div className="flex items-center gap-4">
          <div className="size-20 rounded-full overflow-hidden bg-white/10 border border-white/10">
            {url ? <Image src={url} alt="" width={80} height={80} className="size-full object-cover" /> : null}
          </div>
          <label className="text-sm cursor-pointer px-3 py-1.5 rounded-lg border border-white/10 bg-white/5 hover:bg-white/10">
            {busy ? 'Enviando...' : 'Trocar avatar'}
            <input type="file" accept="image/jpeg,image/png,image/webp" className="hidden" onChange={handleFile} disabled={busy} />
          </label>
        </div>
      )
    }
    ```

    Create `src/app/(auth)/settings/actions.ts` — re-uses updateProfileAction shape but accepts more fields and does NOT redirect:
    ```ts
    'use server'
    import { revalidatePath } from 'next/cache'
    import { createClient } from '@/lib/supabase/server'
    import { usernameSchema } from '@/lib/validation/auth'

    type FormResult = { ok: true } | { ok: false; error: string; fieldErrors?: Record<string, string> }

    export async function saveSettingsAction(formData: FormData): Promise<FormResult> {
      const parsed = usernameSchema.safeParse({
        username: formData.get('username'),
        display_name: formData.get('display_name') || undefined,
        bio: formData.get('bio') || undefined,
        avatar_url: formData.get('avatar_url') || undefined,
      })
      if (!parsed.success) {
        const fieldErrors: Record<string, string> = {}
        for (const issue of parsed.error.issues) fieldErrors[issue.path.join('.')] = issue.message
        return { ok: false, error: 'Dados inválidos', fieldErrors }
      }
      const supabase = await createClient()
      const { data: { user } } = await supabase.auth.getUser()
      if (!user) return { ok: false, error: 'Não autenticado' }

      const { error } = await supabase
        .from('profiles')
        .update({
          username: parsed.data.username,
          display_name: parsed.data.display_name ?? null,
          bio: parsed.data.bio ?? null,
          avatar_url: parsed.data.avatar_url || null,
          updated_at: new Date().toISOString(),
        })
        .eq('id', user.id)

      if (error) {
        if (error.code === '23505') return { ok: false, error: 'Username já em uso', fieldErrors: { username: 'Já está em uso' } }
        return { ok: false, error: error.message }
      }
      revalidatePath('/settings')
      revalidatePath(`/u/${parsed.data.username}`)
      return { ok: true }
    }
    ```
