    Create `src/lib/supabase/middleware.ts`:
    ```ts
    import { createServerClient } from '@supabase/ssr'
    import { NextResponse, type NextRequest } from 'next/server'

    const PUBLIC_PATTERNS: RegExp[] = [
      /^\/$/,
      /^\/u\/[^/]+\/?$/,           // public profile
      /^\/p\/[^/]+\/?$/,           // public prompty (Phase 2)
      /^\/auth\/(login|signup|reset-password|callback)/,
      /^\/api\/public\//,
    ]

    export function isPublicRoute(pathname: string): boolean {
      return PUBLIC_PATTERNS.some((re) => re.test(pathname))
    }

    export async function updateSession(request: NextRequest) {
      let supabaseResponse = NextResponse.next({ request })

      const supabase = createServerClient(
        process.env.NEXT_PUBLIC_SUPABASE_URL!,
        process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
        {
          cookies: {
            getAll() { return request.cookies.getAll() },
            setAll(cookiesToSet) {
              cookiesToSet.forEach(({ name, value }) =>
                request.cookies.set(name, value))
              supabaseResponse = NextResponse.next({ request })
              cookiesToSet.forEach(({ name, value, options }) =>
                supabaseResponse.cookies.set(name, value, options))
            },
          },
        }
      )

      // IMPORTANT: getUser() — never getSession() — for routing decisions
      const { data: { user } } = await supabase.auth.getUser()

      const pathname = request.nextUrl.pathname
      if (!user && !isPublicRoute(pathname)) {
        const url = request.nextUrl.clone()
        url.pathname = '/auth/login'
        url.searchParams.set('next', pathname)
        return NextResponse.redirect(url)
      }

      return supabaseResponse
    }
    ```

    Create `src/lib/images/compress.ts` (covers INFR-03):
    ```ts
    import imageCompression from 'browser-image-compression'

    export const DEFAULT_OPTIONS = {
      maxSizeMB: 0.2,            // 200 KB
      maxWidthOrHeight: 1024,
      fileType: 'image/webp' as const,
      useWebWorker: true,
      initialQuality: 0.85,
    }

    export async function compressAvatar(file: File): Promise<File> {
      if (file.size > 2 * 1024 * 1024) {
        throw new Error('File exceeds 2MB input limit')
      }
      const compressed = await imageCompression(file, DEFAULT_OPTIONS)
      return new File([compressed], file.name.replace(/\.[^.]+$/, '.webp'), {
        type: 'image/webp',
      })
    }

    // Expose defaults for unit tests
    ;(compressAvatar as unknown as { DEFAULT_OPTIONS: typeof DEFAULT_OPTIONS }).DEFAULT_OPTIONS = DEFAULT_OPTIONS
    ```
