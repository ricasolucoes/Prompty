    ```js
    #!/usr/bin/env node
    /**
     * Reads Supabase project usage from the Management API and fails the
     * GitHub Action when any metric crosses 70% (warn) or 90% (critical)
     * of the documented free-tier limit.
     *
     * Required env vars:
     *   SUPABASE_ACCESS_TOKEN  Personal Access Token
     *   SUPABASE_PROJECT_REF   Project reference id
     *
     * Free-tier limits (Supabase, validated 2026-05-06; update as needed):
     *   db_size_bytes:    500 MB  =  524288000
     *   storage_bytes:    1 GB    = 1073741824
     *   egress_bytes:     5 GB    = 5368709120 (per month)
     *   realtime_concurrent_conns: 200
     */
    const LIMITS = {
      db_size_bytes:               524_288_000,
      storage_bytes:             1_073_741_824,
      egress_bytes:              5_368_709_120,
      realtime_concurrent_conns:           200,
    }

    const TOKEN = process.env.SUPABASE_ACCESS_TOKEN
    const REF   = process.env.SUPABASE_PROJECT_REF
    if (!TOKEN || !REF) {
      console.error('Missing SUPABASE_ACCESS_TOKEN or SUPABASE_PROJECT_REF env vars')
      process.exit(2)
    }

    async function fetchUsage() {
      const url = `https://api.supabase.com/v1/projects/${REF}/usage`
      const r = await fetch(url, { headers: { Authorization: `Bearer ${TOKEN}` } })
      if (!r.ok) {
        console.error(`Management API returned ${r.status}: ${await r.text()}`)
        process.exit(2)
      }
      return r.json()
    }

    function metricValue(usage, key) {
      // Tolerate different shapes from the API (it has changed historically)
      if (typeof usage[key] === 'number') return usage[key]
      if (usage.usage && typeof usage.usage[key] === 'number') return usage.usage[key]
      if (Array.isArray(usage)) {
        const m = usage.find(u => u.metric === key || u.name === key)
        if (m) return m.usage ?? m.value ?? 0
      }
      return 0
    }

    const usage = await fetchUsage()

    let breach = 0
    const rows = []
    for (const [key, limit] of Object.entries(LIMITS)) {
      const v = metricValue(usage, key)
      const pct = limit > 0 ? (v / limit) * 100 : 0
      let status = 'OK'
      if (pct >= 90) { status = 'CRITICAL'; breach = Math.max(breach, 2) }
      else if (pct >= 70) { status = 'WARN'; breach = Math.max(breach, 1) }
      rows.push({ key, value: v, limit, pct: pct.toFixed(1) + '%', status })
    }

    console.log('Supabase usage report:')
    console.table(rows)

    if (breach >= 2) {
      console.error('\n::error::CRITICAL — Supabase free-tier usage >= 90%')
      process.exit(1)
    }
    if (breach >= 1) {
      console.warn('\n::warning::WARN — Supabase free-tier usage >= 70%')
      process.exit(1)
    }
    console.log('\nAll metrics under 70%.')
    ```

    ```yaml
    name: Supabase Usage Monitor

    on:
      schedule:
        - cron: '0 9 * * 1'   # Every Monday at 09:00 UTC
      workflow_dispatch:

    permissions:
      contents: read

    jobs:
      check-usage:
        runs-on: ubuntu-latest
        timeout-minutes: 5
        steps:
          - name: Checkout
            uses: actions/checkout@v4

          - name: Setup Node
            uses: actions/setup-node@v4
            with:
              node-version: '20'

          - name: Run usage check
            env:
              SUPABASE_ACCESS_TOKEN: ${{ secrets.SUPABASE_ACCESS_TOKEN }}
              SUPABASE_PROJECT_REF:  ${{ secrets.SUPABASE_PROJECT_REF }}
            run: node scripts/check-limits.mjs
    ```

    ```markdown
    # Supabase Free-Tier Usage Monitoring

    Implements requirement INFR-05: weekly cron alerts at 70% and 90% of free-tier limits.

    ## How it works

    1. `.github/workflows/supabase-monitor.yml` runs every Monday at 09:00 UTC.
    2. The workflow runs `scripts/check-limits.mjs`, which calls the Supabase

       - `storage_bytes`: 1 GB
       - `egress_bytes`: 5 GB / month
       - `realtime_concurrent_conns`: 200
    4. If any metric crosses 70%, the workflow fails — GitHub emails the repo

    | `SUPABASE_ACCESS_TOKEN` | https://supabase.com/dashboard/account/tokens |
    | `SUPABASE_PROJECT_REF`  | Project Settings -> General -> Reference ID |
    | `ALERT_EMAIL`           | Optional — email to surface in logs |

    2. Trigger manually via `Actions -> Supabase Usage Monitor -> Run workflow`.
    3. Confirm the run reports a usage table and exits with status 0 (under 70%).
    4. Confirm GitHub workflow notifications are enabled for the repo owner so
       failure emails are delivered.

    ## Known caveat

    The Management API endpoint shape is NOT formally versioned by Supabase.
    ```
