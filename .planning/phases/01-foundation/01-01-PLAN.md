    ```sql
    -- RLS role tests — run via: psql $DATABASE_URL -f supabase/tests/rls_roles.sql
    -- Each block must succeed (no error) for the test to pass.

    BEGIN;

    -- INFR-01: every table in public schema has rowsecurity = true
    DO $$
    DECLARE r record; missing text := '';
    BEGIN
      FOR r IN SELECT tablename FROM pg_tables WHERE schemaname = 'public' LOOP
        IF NOT (SELECT relrowsecurity FROM pg_class WHERE relname = r.tablename AND relnamespace = 'public'::regnamespace) THEN
          missing := missing || r.tablename || ', ';
        END IF;
      END LOOP;
      IF length(missing) > 0 THEN RAISE EXCEPTION 'Tables missing RLS: %', missing; END IF;
    END $$;

    -- INFR-02 / GAME-02: authenticated role cannot INSERT into point_events
    SET ROLE authenticated;
    DO $$
    BEGIN
      BEGIN
        INSERT INTO point_events (user_id, event_type, reference_id, points)
          VALUES ('00000000-0000-0000-0000-000000000001'::uuid, 'manual_fraud', null, 9999);
        RAISE EXCEPTION 'INFR-02 FAIL: authenticated role was able to INSERT into point_events';
      EXCEPTION WHEN insufficient_privilege OR check_violation THEN
        -- expected
        NULL;
      END;
    END $$;
    RESET ROLE;

    ROLLBACK;
    ```
